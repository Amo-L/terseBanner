.banner-thumb-auto .eb-thumb ul li{
    margin-right: 10px;
    border: 2px solid transparent;
    opacity: .7;
    filter: alpha(opactiy=70);
}
.banner-thumb-auto .eb-thumb ul li.active{
    border-color: #00b579;
    opacity: 1;
    filter: alpha(opactiy=100);
}


$('.banner-thumb-auto').easyBanner({
    serial: 'thumb'
});

Set thumbnail manually.

  • HTML
  • CSS
  • JAVASCRIPT

<div class="banner-thumb-manual">
    <ul>
        // add "?thumb=['thumbnail url']" after img src
        <li><img src="img/banner-1.jpg?thumb=img/thumb-1.jpg"></li>
        <li><img src="img/banner-2.jpg?thumb=img/thumb-2.jpg"></li>
        <li><img src="img/banner-3.jpg?thumb=img/thumb-3.jpg"></li>
        <li><img src="img/banner-4.jpg?thumb=img/thumb-4.jpg"></li>
        <li><img src="img/banner-5.jpg?thumb=img/thumb-5.jpg"></li>
    </ul>
</div>

}
.banner-thumb-manual .eb-thumb ul li.active{
    border-color: #00b579;
